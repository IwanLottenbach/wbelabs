// Aufgabe 4a
const fibo = function(n){
    
    if (n <= 1){
        return n; 
    }

    return fibo(n-1) + fibo(n-2);
  }

// Aufgabe 4a fakultativ
const fiboIterative = function(n){
    
    if (n <= 1){
        return n; 
    }

    let n1 = 1
    let n2 = 0
    let result = 0
    
    for(let i = 2; i <(n+1);i++){

        result = n1 + n2
        n2 = n1
        n1 = result

    }
  
    return result;
}

const fiboGoldenerSchnitt = function(n){
    let G = (1+Math.sqrt(5))/2

    if (n <= 1){
        return n; 
    }

    return (G**n)/Math.sqrt(5)
  
}
